    }
    
    fun compareORGate(x:Boolean, y:Boolean): Boolean {
        if (x === true) {
            return true
        }
        if (y === true) {
            return true
        }
        
        return false
    }
    
    fun comapreANDGate(x: Boolean, y: Boolean): Boolean {
        if (x === true && y === true) {
            return true
        } else {
            return false
        }
    }
}
